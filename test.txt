-- -----------------------------------------------------------------------------
-- Temporary table name: var_logged_30D_tmp01
-- accesses_30D
-- number_used_platforms_30D
-- number_used_devices_30D
-- -----------------------------------------------------------------------------

INSERT OVERWRITE TABLE var_logged_30D_tmp01
select
    vlm.date,
    vlm.user_id,
    vlm.birth_date,
    vlm.gender,
    vlm.age_bracket,
    vlm.country,
    vlm.state,
    vlm.province,
    vlm.city,
    vlm.zip_code,
    vlm.registration_method,
    vlm.referral_source,
    vlm.registration_trigger,
    vlm.registration_date,
    vlm.most_used_platform,
    vlm.first_access_date,
    vlm.first_view_date,
    tmp.accesses_30D,
    tmp.number_used_platforms_30D,
    tmp.number_used_devices_30D
from
    var_logged_master vlm, (select
    user_id,
    count(accesses_30D) AS accesses_30D,
    count(distinct(number_used_platforms_30D)) AS number_used_platforms_30D,
    count(distinct(number_used_devices_30D)) AS number_used_devices_30D

from (

select

    final.user_id1 AS user_id,
    final.user_id2 AS accesses_30D,
    final.channel AS number_used_platforms_30D,
    final.deviceid AS number_used_devices_30D

from (
select t1.user_id AS user_id1, t2.userid AS user_id2, t2.channel, t2.deviceid
from
    var_logged_master t1
    left outer join
    (select * from target_login t3 where  t3.dt_reference >= DATE_SUB(FROM_UNIXTIME(UNIX_TIMESTAMP(),'yyyy-MM-dd') , 30) 
      AND t3.dt_reference < FROM_UNIXTIME(UNIX_TIMESTAMP(),'yyyy-MM-dd')) t2
    on t1.user_id = t2.userid) final) a
group by user_id) tmp
where vlm.user_id = tmp.user_id;

    var_logged_master t1
    left outer join
    (select * from target_impression t3 where t3.dt_reference >= DATE_SUB(FROM_UNIXTIME(UNIX_TIMESTAMP(),'yyyy-MM-dd') , 30) 
      AND t3.dt_reference < FROM_UNIXTIME(UNIX_TIMESTAMP(),'yyyy-MM-dd')) t2

    var_logged_master t1

-- -----------------------------------------------------------------------------
-- Temporary table name: var_logged_30D_tmp09
-- video_drop_off_30D
-- average_video_drop_off_30D
-- launched_apps_30D
-- -----------------------------------------------------------------------------

INSERT OVERWRITE TABLE var_logged_30D_tmp09
select
    vlm.date,
    vlm.user_id,
    vlm.birth_date,
    vlm.gender,
    vlm.age_bracket,
    vlm.country,
    vlm.state,
    vlm.province,
    vlm.city,
    vlm.zip_code,
    vlm.registration_method,
    vlm.referral_source,
    vlm.registration_trigger,
    vlm.registration_date,
    vlm.most_used_platform,
    vlm.first_access_date,
    vlm.first_view_date,
    tmp.video_drop_off_30D,
    tmp.average_video_drop_off_30D,
    tmp.launched_apps_30D
from
    var_logged_master vlm, (select
    user_id,
    count(video_drop_off_30D) AS video_drop_off_30D,
    (count(average_video_drop_off_30D_num1)/count(average_video_drop_off_30D_num2)) AS average_video_drop_off_30D,
    count(launched_apps_30D) AS launched_apps_30D

from (

select

    final.user_id1 AS user_id,
    (if(final.eventtype == "VIDEOEND" and final.terminationreason == "ERROR", final.sessionid, null)) AS video_drop_off_30D,
    (if(final.eventtype == "VIDEOEND" and final.terminationreason == "ERROR", final.sessionid, null)) AS average_video_drop_off_30D_num1,
    final.user_id2 AS average_video_drop_off_30D_num2,
    (if(final.eventtype == "APPLICATIONSTARTUP", final.sessionid, null)) AS launched_apps_30D

from (
select t1.user_id AS user_id1, t2.userid AS user_id2, t2.sessionid, t2.eventtype, t2.terminationreason
from
    var_logged_master t1
    left outer join
    (select * from target_user_action t3 where t3.dt_reference >= DATE_SUB(FROM_UNIXTIME(UNIX_TIMESTAMP(),'yyyy-MM-dd') , 30) 
      AND t3.dt_reference < FROM_UNIXTIME(UNIX_TIMESTAMP(),'yyyy-MM-dd')) t2
    on t1.user_id = t2.userid) final) a
group by user_id) tmp
where vlm.user_id = tmp.user_id;
